  <script>

    const questionStarters = [

      "Imagine if", "What would happen if", "Describe a world where", "If you could ask", "Pretend that", "Suppose"

    ];

    const subjects = [

      "penguins", "time-traveling sloths", "sentient furniture", "ghosts of forgotten memes", "AI poets", "talking books"

    ];

    const contexts = [

      "formed a rock band?", "ran for president?", "became fashion designers?", "turned invisible?", "spoke only in emojis?"

    ];

    const twists = [

      "in space", "during a thunderstorm", "at a tea party", "inside a dream", "on live TV", "while upside down"

    ];


    function getRandomItem(arr) {

      return arr[Math.floor(Math.random() * arr.length)];

    }


    function animateReel(id) {

      const el = document.getElementById(id);

      el.classList.add('spinning');

      setTimeout(() => el.classList.remove('spinning'), 500);

    }


    function reroll(reelId) {

      const map = {

        'reel1': questionStarters,

        'reel2': subjects,

        'reel3': contexts,

        'reel4': twists

      };

      const newText = getRandomItem(map[reelId]);

      document.getElementById(reelId).textContent = newText;

      animateReel(reelId);

      updateResult();

    }


    function spinReels() {

      document.getElementById('reel1').textContent = getRandomItem(questionStarters);

      document.getElementById('reel2').textContent = getRandomItem(subjects);

      document.getElementById('reel3').textContent = getRandomItem(contexts);

      document.getElementById('reel4').textContent = getRandomItem(twists);

      ['reel1','reel2','reel3','reel4'].forEach(animateReel);

      updateResult();

    }


    function updateResult() {

      const q = `${reel1.textContent} ${reel2.textContent} ${reel3.textContent} ${reel4.textContent}`;

      document.getElementById('result').textContent = q;

    }


    function surpriseMe() {

      spinReels();

      updateResult();

    }


    function sendToGemini() {

      const prompt = document.getElementById('result').textContent;

      const url = `https://gemini.google.com/app?prompt=${encodeURIComponent(prompt)}`;

      window.open(url, '_blank');

    }


    // Event listener

    document.getElementById('spinButton').addEventListener('click', spinReels);

  </script>
