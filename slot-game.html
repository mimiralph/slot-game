    <script>

        // --- REVISED Data Arrays for Fun, Relaxed Questions ---

        const questionStarters = [

            "Imagine if", "What would happen if", "Describe a world where", "What's the secret story behind",

            "If you could ask", "Pretend that", "What funny reason explains why", "Suppose",

            "Invent a holiday celebrating", "What if", "Tell a tall tale about"

        ];


        const subjects = [

            "dinosaurs", "garden gnomes", "talking squirrels", "lost socks", "clouds",

            "friendly monsters under the bed", "pet rocks", "fairies", "a mischievous teapot",

            "bicycles", "books", "stars", "singing flowers", "tiny dragons", "your reflection"

        ];


        const contexts = [

            "learned to skateboard?", "started a band?", "could grant wishes, but only silly ones?",

            "secretly ran the internet?", "threw a surprise party?", "decided to become astronauts?",

            "wore tiny hats?", "could only communicate through riddles?", "were afraid of butterflies?",

            "built elaborate sandcastles overnight?", "could change color based on their mood?",

            "held weekly meetings in your attic?", "tried to learn magic tricks?", "went on vacation to the moon?"

        ];


        // --- Element References ---

        const reel1 = document.getElementById('reel1');

        const reel2 = document.getElementById('reel2');

        const reel3 = document.getElementById('reel3');

        const spinButton = document.getElementById('spinButton');

        const resultQuestion = document.getElementById('result');


        // --- Helper Function: Get Random Item ---

        function getRandomItem(arr) {

            const randomIndex = Math.floor(Math.random() * arr.length);

            return arr[randomIndex];

        }


        // --- Helper Function: Animate Reel ---

        function animateReel(reelElement) {

            reelElement.classList.add('spinning');

            // Remove the class after the animation completes

            setTimeout(() => {

                reelElement.classList.remove('spinning');

            }, 500); // Match animation duration

        }



        // --- Spin Function ---

        function spinReels() {

            // Get random items from the fun arrays

            const selectedStarter = getRandomItem(questionStarters);

            const selectedSubject = getRandomItem(subjects);

            const selectedContext = getRandomItem(contexts);


            // Display items in reels with animation

            reel1.textContent = selectedStarter;

            animateReel(reel1);


            reel2.textContent = selectedSubject;

            animateReel(reel2);


            reel3.textContent = selectedContext;

            animateReel(reel3);


            // Construct and display the final question

            resultQuestion.textContent = `${selectedStarter} ${selectedSubject} ${selectedContext}`;

        }


        // --- Event Listener ---

        spinButton.addEventListener('click', spinReels);


        // --- Initial State ---

        // Keep placeholders initially
